import cv2
import time
import os

def play_binary_video(video_path):
    # Открываем видеофайл
    cap = cv2.VideoCapture(video_path)
    
    if not cap.isOpened():
        print("Не удалось открыть видеофайл.")
        return

    # Получаем FPS для синхронизации скорости
    fps = cap.get(cv2.CAP_PROP_FPS)
    frame_delay = 1.0 / fps if fps > 0 else 1.0 / 30.0

    # Очищаем терминал перед началом
    os.system('cls' if os.name == 'nt' else 'clear')

    while True:
        start_time = time.time()
        ret, frame = cap.read()
        
        if not ret:
            break  # Видео закончилось

        # Пытаемся получить текущий размер терминала
        try:
            term_cols, term_lines = os.get_terminal_size()
        except OSError:
            term_cols, term_lines = 80, 24

        # Учитываем высоту шрифта (символы обычно вытянуты по вертикали)
        # Оставляем 1 строку для избежания сдвига вывода
        width = term_cols
        height = term_lines - 1 

        # Меняем размер кадра и переводим в градации серого
        resized = cv2.resize(frame, (width, height))
        gray = cv2.cvtColor(resized, cv2.COLOR_BGR2GRAY)

        output = ""
        for row in gray:
            for pixel in row:
                # Порог 127: разделяем на два символа
                output += "1" if pixel > 127 else "0"
            output += "\n"

        # \033[H возвращает курсор в левый верхний угол (без мерцания)
        print("\033[H" + output, end="")

        # Высчитываем время для поддержания правильного FPS
        elapsed = time.time() - start_time
        sleep_time = frame_delay - elapsed
        if sleep_time > 0:
            time.sleep(sleep_time)

    cap.release()
    print("\nВоспроизведение завершено.")

# Запуск скрипта
# Укажи путь к своему mp4 файлу
play_binary_video("video.mp4")
