with open('story.txt', 'rb') as f:
    data = f.read()

# Вытаскиваем все не-ASCII байты
hidden_bytes = b''
for byte in data:
    if byte > 127:
        hidden_bytes += bytes([byte])

# Разбиваем на 3-байтовые последовательности (стандартный размер таких юникод-символов)
chars = [hidden_bytes[i:i+3].hex() for i in range(0, len(hidden_bytes), 3) if len(hidden_bytes[i:i+3]) == 3]

# Находим уникальные символы
unique_chars = list(set(chars))
print(f"Найдено уникальных невидимых символов: {len(unique_chars)}")
print(f"Алфавит: {unique_chars}")

if len(unique_chars) == 2:
    print("\nПробуем два варианта бинарной расшифровки...\n")
    char0, char1 = unique_chars[0], unique_chars[1]
    
    # Вариант 1 (первый символ - 0, второй - 1)
    bin1 = "".join(['0' if c == char0 else '1' for c in chars])
    # Вариант 2 (наоборот)
    bin2 = "".join(['1' if c == char0 else '0' for c in chars])

    for i, b_str in enumerate([bin1, bin2]):
        try:
            # Переводим биты в текст
            res = "".join([chr(int(b_str[j:j+8], 2)) for j in range(0, len(b_str), 8)])
            print(f"--- Вариант {i+1} ---")
            print(res)
        except Exception:
            pass
