from pwn import *
import re

# Замени IP и порт, если инстанс перезапускался
HOST = '154.57.164.63'
PORT = 32073

def get_signature(p, hex_msg):
    p.sendlineafter(b"> ", b"1")
    p.sendlineafter(b"hex: ", hex_msg.encode())
    p.recvuntil(b"Token issued: ")
    
    # Читаем вплоть до закрывающей скобки массива
    token_data = p.recvuntil(b"]").decode()
    
    # Регулярка для извлечения только hex-символов (игнорируя 0x)
    clean_tokens = []
    for m in re.finditer(r'(?:0x)?([0-9a-fA-F]+)', token_data):
        # Очищаем, переводим в нижний регистр и добиваем нулями
        val = m.group(1).lower().zfill(64)
        if len(val) == 64:
            clean_tokens.append(val)
    
    if len(clean_tokens) < 256:
        log.error(f"Получено {len(clean_tokens)} токенов. Ожидалось 256.")
        
    # Возвращаем ровно 256 частей
    return clean_tokens[:256]

def main():
    p = remote(HOST, PORT)

    log.info("Вытягиваем 'нулевые' части ключа...")
    zero_keys = get_signature(p, "00" * 32)
    
    log.info("Вытягиваем 'единичные' части ключа...")
    one_keys = get_signature(p, "ff" * 32)

    log.info("Формируем подпись для d9_netadmin...")
    target = b"d9_netadmin"
    padded_target = target.rjust(32, b'\x00')

    target_bits = []
    for byte in padded_target:
        for i in range(7, -1, -1):
            target_bits.append((byte >> i) & 1)

    forged_token = []
    for i in range(256):
        if target_bits[i] == 1:
            forged_token.append(one_keys[i])
        else:
            forged_token.append(zero_keys[i])

    log.info("Отправляем поддельную подпись серверу...")
    p.sendlineafter(b"> ", b"2")
    p.sendlineafter(b"validate: ", target)
    
    payload = ",".join(forged_token)
    p.sendlineafter(b"(comma-separated hex): ", payload.encode())
    
    log.success("Запрос отправлен. Читаем ответ сервера...")
    
    try:
        response = p.recvall(timeout=5).decode()
        print(response)
    except EOFError:
        print("[-] Сервер снова разорвал соединение.")

if __name__ == "__main__":
    main()
