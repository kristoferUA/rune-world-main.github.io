def get_bits(data):
    # Хэшируем сообщение и переводим в массив из 256 бит (от старшего к младшему)
    h = hashlib.sha256(data).digest()
    return [(b >> i) & 1 for b in h for i in range(7, -1, -1)]

def generate_covering_payloads(target_msg):
    target_bits = get_bits(target_msg)
    log.info("Начинаем локальный подбор покрывающих сообщений...")
    
    # Крутим цикл, пока не уложимся в лимит из 6 сообщений
    while True:
        payloads = []
        source_map = {} # Запоминаем, из какого пэйлоада берем нужный бит
        covered = [False] * 256
        
        while not all(covered):
            best_msg = None
            best_cover_count = -1
            best_msg_bits = None
            
            # Генерируем 5000 случайных строк и выбираем ту, которая закроет больше всего недостающих бит
            for _ in range(5000):
                msg = os.urandom(8)
                bits = get_bits(msg)
                cover_count = sum(1 for i in range(256) if not covered[i] and bits[i] == target_bits[i])
                
                if cover_count > best_cover_count:
                    best_cover_count = cover_count
                    best_msg = msg
                    best_msg_bits = bits
                    
            p_idx = len(payloads)
            payloads.append(best_msg)
            
            # Отмечаем биты, которые мы смогли добыть
            for i in range(256):
                if not covered[i] and best_msg_bits[i] == target_bits[i]:
                    covered[i] = True
                    source_map[i] = p_idx
                    
        if len(payloads) <= 6:
            log.success(f"Найдено оптимальное покрытие за {len(payloads)} сообщений!")
            return payloads, source_map, target_bits

def get_signature(p, hex_msg):
    p.sendlineafter(b"> ", b"1")
    p.sendlineafter(b"hex: ", hex_msg.encode())
    p.recvuntil(b"Token issued: ")
    
    token_data = p.recvuntil(b"]").decode()
    
    clean_tokens = []
    for m in re.finditer(r'(?:0x)?([0-9a-fA-F]+)', token_data):
        val = m.group(1).lower().zfill(64)
        if len(val) == 64:
            clean_tokens.append(val)
            
    return clean_tokens[:256]
