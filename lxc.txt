from pwn import *
from Crypto.Util.number import long_to_bytes, bytes_to_long
from hashlib import sha256

HOST = '154.57.164.67'
PORT = 30098

def main():
    p = remote(HOST, PORT)
    
    p.recvuntil(b"Ledger hash parameters are a = ")
    a = int(p.recvuntil(b",", drop=True))
    p.recvuntil(b"b = ")
    b = int(p.recvuntil(b" ", drop=True))
    p.recvuntil(b"n = ")
    n = int(p.recvline())
    
    log.info(f"Параметры перехвачены: n = {n}")

    genesis_block = sha256(b"Korvia command channel genesis").digest()
    last_hash = (a * bytes_to_long(genesis_block) + b) % n

    log.info("Начинаем майнинг блоков алгебраическим методом...")
    
    for i in range(100):
        p.recvuntil(b"Validate proxy transaction batch:\n")
        
        blockdata = b""
        for _ in range(3):
            blockdata += p.recvline()
            
        P = long_to_bytes(last_hash)
        X = bytes_to_long(P + blockdata)
        
        h = 1
        N = (pow(a, -1, n) * (h - b - a * X * (1 << 256))) % n
        
        nonce_hex = hex(N)[2:].zfill(64)
        p.sendlineafter(b"Enter block nonce in hex: ", nonce_hex.encode())
        
        # Читаем ответ сервера на отправленный блок
        res = p.recvline().decode()
        
        if "[REJECTED]" in res:
            log.error("Блок отклонен! Проверь математику.")
            break
        elif "Flag payload:" in res:
            # Ловим флаг прямо здесь, на 100-й итерации
            log.success("Получен хэш флага!")
            flag_hash_hex = res.split("Flag payload: ")[1].strip()
            flag_hash = int(flag_hash_hex, 16)
            
            log.info("Оборачиваем хэш-функцию для расшифровки флага...")
            F = (pow(a, -1, n) * (flag_hash - b)) % n
            flag = long_to_bytes(F).decode()
            print(f"\n[+] Взлом завершен. Флаг: {flag}")
            return
            
        last_hash = h
        
        if (i + 1) % 10 == 0:
            log.success(f"Пройдено блоков: {i + 1}/100")

if __name__ == "__main__":
    main()
