def bits_to_ascii_padding(path_bits):
    # Превращаем биты изогений в валидные ASCII-символы (от 0x20 до 0x7E)
    # Чтобы не сломать eval, используем только печатные символы
    padding = bytearray()
    
    # Дополняем биты нулями, чтобы их длина была кратна 8
    bits = list(path_bits)
    while len(bits) % 8 != 0:
        bits.append(0)
        
    for i in range(0, len(bits), 8):
        byte_chunk = bits[i:i+8]
        # Так как нам нужен валидный ASCII, стартовый бит всегда должен быть 0
        # Если KLPT требует 1 первым битом, придется сделать шаг назад в графе
        # или использовать пробелы/табы хитрее. 
        # В простейшем случае:
        byte_val = int("".join(map(str, byte_chunk)), 2)
        
        # Если значение выходит за рамки ASCII, можно использовать 
        # эквивалентные битовые последовательности (циклы в графе), 
        # чтобы перестроить путь под печатные символы.
        padding.append(byte_val)
        
    return padding

def deterministic_sqrt(val):
    sqrts = val.sqrt(all=True)
    return min(sqrts) if sqrts else None

def select(lambda0, lambda1, b):
    return max(lambda0, lambda1) if b else min(lambda0, lambda1)

def next_curve(A, alpha, b):
    xi = alpha**2
    zeta = 3 * xi + A
    eta = deterministic_sqrt(zeta)
    new_A = -(4 * A + 15 * xi)
    lambda0 = alpha + 2 * eta
    lambda1 = alpha - 2 * eta
    return new_A, select(lambda0, lambda1, b)

def bits_to_utf8_padding(path_bits):
    padding = bytearray()
    bits = list(path_bits)
    while len(bits) % 8 != 0:
        bits.append(0)
        
    for i in range(0, len(bits), 8):
        byte_val = int("".join(map(str, bits[i:i+8])), 2)
        # Ограничиваем безопасным ASCII диапазоном (пробелы, табы, буквы)
        padding.append(byte_val & 0x7F) 
    return padding
