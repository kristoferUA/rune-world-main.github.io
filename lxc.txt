from pwn import *

# Замени на данные из запущенного докера на HTB
HOST = '127.0.0.1'
PORT = 1337

def get_signature(p, hex_msg):
    p.sendlineafter(b"> ", b"1")
    p.sendlineafter(b"hex: ", hex_msg.encode())
    p.recvuntil(b"Token issued: [")
    token_str = p.recvline().strip().decode()[:-1]
    
    # Очищаем от мусора, кавычек и префиксов 0x если они есть
    clean_tokens = []
    for x in token_str.split(','):
        val = x.strip().replace('"', '').replace("'", "")
        if val.startswith('0x'):
            val = val[2:]
        clean_tokens.append(val)
    return clean_tokens

def main():
    p = remote(HOST, PORT)

    print("[*] Вытягиваем 'нулевые' части ключа...")
    zero_keys = get_signature(p, "00" * 32)
    
    print("[*] Вытягиваем 'единичные' части ключа...")
    one_keys = get_signature(p, "ff" * 32)

    print("[*] Формируем подпись для d9_netadmin...")
    target = b"d9_netadmin"
    # Дополняем нулями до 32 байт (256 бит), как это делает Rust скрипт
    padded_target = target.rjust(32, b'\x00')

    target_bits = []
    for byte in padded_target:
        for i in range(7, -1, -1):
            target_bits.append((byte >> i) & 1)

    forged_token = []
    for i in range(256):
        if target_bits[i] == 1:
            forged_token.append(one_keys[i])
        else:
            forged_token.append(zero_keys[i])

    print("[*] Отправляем поддельную подпись серверу...")
    p.sendlineafter(b"> ", b"2")
    p.sendlineafter(b"validate: ", target)
    
    # Сервер ожидает токены через запятую в hex-формате
    payload = ",".join(forged_token)
    p.sendlineafter(b"(comma-separated hex): ", payload.encode())
    
    # Получаем флаг
    print(p.recvall().decode())

if __name__ == "__main__":
    main()
