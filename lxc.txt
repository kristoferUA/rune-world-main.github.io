from sage.all import GF, PolynomialRing

# Инициализация поля из оригинального скрипта
p = 2**49 * 3**36 - 1
K = GF(p**2, "x", modulus=[1, 0, 1])
x = K.gen()
R = PolynomialRing(K, 'X')
X_var = R.gen()

def deterministic_sqrt(val):
    return min(val.sqrt(all=True))

def select(lambda0, lambda1, b):
    if b:
        return max(lambda0, lambda1)
    else:
        return min(lambda0, lambda1)

def next_curve(A, alpha, b):
    xi = alpha**2
    zeta = 3 * xi + A
    eta = deterministic_sqrt(zeta)
    new_A = -(4 * A + 15 * xi)

    lambda0 = alpha + 2 * eta
    lambda1 = alpha - 2 * eta
    new_alpha = select(lambda0, lambda1, b)
    return new_A, new_alpha

def prev_curve(new_A, new_alpha, b):
    # Квадратное уравнение: 16*X^2 + (4*new_alpha^2 + 8*new_A)*X + (new_alpha^2 + new_A)^2 = 0
    c2 = K(16)
    c1 = K(4 * new_alpha**2 + 8 * new_A)
    c0 = K((new_alpha**2 + new_A)**2)
    
    poly = c2 * X_var**2 + c1 * X_var + c0
    roots = poly.roots(multiplicities=False)
    
    for xi in roots:
        # Восстанавливаем A из формулы: new_A = -(4*A + 15*xi)
        A_cand = (-new_A - 15 * xi) / K(4)
        
        try:
            alphas = xi.sqrt(all=True)
        except ValueError:
            continue # Если корня нет в поле, пропускаем ветку
            
        for alpha_cand in alphas:
            # Делаем проверочный шаг вперед, чтобы убедиться в правильности выбора
            test_A, test_alpha = next_curve(A_cand, alpha_cand, b)
            if test_A == new_A and test_alpha == new_alpha:
