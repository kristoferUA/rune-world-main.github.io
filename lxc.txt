def generate_nec_packet(address, command):
    inv_address = (~address) & 0xFF
    inv_command = (~command) & 0xFF
    return bytes([address, inv_address, command, inv_command])

def send_packet(payload, send_as_text=False):
    try:
        with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as s:
            s.settimeout(1.0)  # Даем сокету до 1 секунды на коннект
            s.connect((HOST, IR_PORT))
            if send_as_text:
                s.sendall(payload.hex().encode() + b'\n')
            else:
                s.sendall(payload)
            time.sleep(0.05)  # Небольшая пауза, чтобы бэкенд успел переварить байты
    except:
        pass

import glob

def get_blocks_from_section(section, global_cx, global_cz, sy):
    found = []
    # В новых версиях Minecraft (1.16+) состояния лежат в block_states
    if "block_states" not in section:
        return found
        
    block_states = section["block_states"]
    if "palette" not in block_states:
        return found
        
    palette = block_states["palette"]
    
    # Сначала проверяем, есть ли вообще в этой секции нужные нам блоки
    interesting = False
    for p in palette:
        name = p["Name"].value if "Name" in p else p["name"].value
        if name in ["minecraft:lever", "minecraft:observer", "minecraft:redstone_lamp"]:
            interesting = True
            break
            
    # Если блоков нет или секция пустая - скипаем (экономит 99% времени)
    if not interesting or "data" not in block_states:
        return found
        
    # Распаковка битового массива 
    data_array = block_states["data"]
    palette_len = len(palette)
    bits_per_block = max(4, math.ceil(math.log2(palette_len)))
    blocks_per_long = 64 // bits_per_block
    mask = (1 << bits_per_block) - 1
    
    for i in range(4096):
        long_index = i // blocks_per_long
        if long_index >= len(data_array):
            break
        long_val = data_array[long_index].value
        
        # Java longs знаковые, переводим в беззнаковые
        if long_val < 0:
            long_val += (1 << 64)
            
        bit_offset = (i % blocks_per_long) * bits_per_block
        palette_index = (long_val >> bit_offset) & mask
        
        if palette_index < palette_len:
            block_prop = palette[palette_index]
            name = block_prop["Name"].value if "Name" in block_prop else block_prop["name"].value
            
            if name in ["minecraft:lever", "minecraft:observer", "minecraft:redstone_lamp"]:
                # Вычисляем локальные координаты внутри 16x16x16
                y = (i >> 8) & 0xF
                z = (i >> 4) & 0xF
                x = i & 0xF
                
                # Переводим в глобальные координаты мира
                gx = global_cx * 16 + x
                gy = sy * 16 + y
                gz = global_cz * 16 + z
                
                # Вытягиваем свойства (powered, lit, facing)
                props = {}
                if "Properties" in block_prop:
                    for k, v in block_prop["Properties"].items():
                        props[k] = v.value
                        
                found.append((name, (gx, gy, gz), props))
                
    return found
