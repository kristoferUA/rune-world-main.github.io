from sage.all import GF, EllipticCurve
import hashlib
import itertools

p = 2**49 * 3**36 - 1
K = GF(p**2, "x", modulus=[1, 0, 1])
x = K.gen()

def deterministic_sqrt(val):
    return min(val.sqrt(all=True))

def select(lambda0, lambda1, b):
    return max(lambda0, lambda1) if b else min(lambda0, lambda1)

def next_curve(A, alpha, b):
    xi = alpha**2
    zeta = 3 * xi + A
    eta = deterministic_sqrt(zeta)
    new_A = -(4 * A + 15 * xi)
    lambda0 = alpha + 2 * eta
    lambda1 = alpha - 2 * eta
    new_alpha = select(lambda0, lambda1, b)
    return new_A, new_alpha

def data_to_bits(data):
    return [int(b) for c in data for b in "{:08b}".format(c)]

def walk_path(A_start, alpha_start, bits):
    A, alpha = A_start, alpha_start
    for b in bits:
        A, alpha = next_curve(A, alpha, b)
    return A, alpha
