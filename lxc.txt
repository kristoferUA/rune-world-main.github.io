from sage.all import GF, EllipticCurve
from hashlib import sha256
import random
from pwn import remote

p = 2**49 * 3**36 - 1
K = GF(p**2, "x", modulus=[1, 0, 1])
x = K.gen()

def deterministic_sqrt(val):
    return min(val.sqrt(all=True))

def select(lambda0, lambda1, b):
    if b: return max(lambda0, lambda1)
    else: return min(lambda0, lambda1)

def next_curve(A, alpha, b):
    xi = alpha**2
    zeta = 3 * xi + A
    eta = deterministic_sqrt(zeta)
    new_A = -(4 * A + 15 * xi)
    lambda0 = alpha + 2 * eta
    lambda1 = alpha - 2 * eta
    new_alpha = select(lambda0, lambda1, b)
    return new_A, new_alpha

def data_to_bits(data):
    if isinstance(data, str): data = data.encode()
    bits = [int(b) for c in data for b in "{:08b}".format(c)]
    return bits

def prev_curve(A_next, alpha_next, b):
    disc = -3 * alpha_next**2 - 4 * A_next
    sqrts = disc.sqrt(all=True)
    if not sqrts: return None
    
    for s in sqrts:
        alpha_cand = (alpha_next + s) / 4
        A_cand = (-A_next - 15 * alpha_cand**2) / 4
        try:
            fA, falpha = next_curve(A_cand, alpha_cand, b)
            if fA == A_next and falpha == alpha_next:
                return A_cand, alpha_cand
        except Exception:
            pass
    return None
