def deterministic_sqrt(val):
    sqrts = val.sqrt(all=True)
    return min(sqrts) if sqrts else None

def select(lambda0, lambda1, b):
    return max(lambda0, lambda1) if b else min(lambda0, lambda1)

def bits_to_utf8_padding(path_bits):
    # Превращаем биты в байты. 
    # Python eval() ожидает валидный UTF-8. 
    # Если KLPT выдает биты, не кратные 8, или байты вне UTF-8,
    # здесь может потребоваться добавить логику backtracking-а 
    # (шаг вперед-назад в графе), чтобы скорректировать биты.
    padding = bytearray()
    bits = list(path_bits)
    
    while len(bits) % 8 != 0:
        bits.append(0)
        
    for i in range(0, len(bits), 8):
        byte_val = int("".join(map(str, bits[i:i+8])), 2)
        # Ограничиваем ASCII диапазоном, чтобы не сломать парсер Python
        # В реальном эксплойте тут нужна тонкая подгонка циклов графа
        padding.append(byte_val & 0x7F) 
    return padding
