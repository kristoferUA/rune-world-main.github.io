import base64

# Читаем скачанную "картинку" как сырые байты
with open("roses.jpg", "rb") as f:
    data = f.read()

marker = b""
idx = data.find(marker)

if idx != -1:
    # Берем зашифрованные данные после маркера и убираем лишние переносы строк
    raw_payload = data[idx + len(marker):].strip()

    # Применяем XOR с ключом ROSES
    key = b"ROSES"
    xor_result = bytearray()
    for i in range(len(raw_payload)):
        xor_result.append(raw_payload[i] ^ key[i % len(key)])
        
    # Декодируем из Base64
    b64_string = xor_result.decode('ascii')
    final_vbs = base64.b64decode(b64_string).decode('utf-8')
    
    # Сохраняем VBScript
    with open("valentine.vbs", "w") as f:
        f.write(final_vbs)
    print("Успех! VBScript извлечен в файл valentine.vbs")
else:
    print("Маркер не найден. Видимо, это просто обычные розы...")
